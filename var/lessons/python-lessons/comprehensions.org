* dict comprehensions

  #+begin_src python
    names = ["Bruce", "Clark", "Peter"]
    heroes = ["Batman", "Superman", "Spiderman"]
    print(list(zip(names,heroes)))
  #+end_src

  #+RESULTS:
  : [('Bruce', 'Batman'), ('Clark', 'Superman'), ('Peter', 'Spiderman')]

** tradional
   #+begin_src python
     myd={}
     for n,h in zip(names, heroes):
         myd[n]=h
     print(myd)
   #+end_src

   #+RESULTS:
   : {'Bruce': 'Batman', 'Clark': 'Superman', 'Peter': 'Spiderman'}

** dict comprehension
   #+begin_src python
     myd={n:h for n,h in zip(names,heroes)}
     print(myd)
   #+end_src

   #+RESULTS:
   : {'Bruce': 'Batman', 'Clark': 'Superman', 'Peter': 'Spiderman'}
